模拟一个 map 方法

```js
Array.prototype.mymap = function (itemfn: Function) {
    var new_arr = []
    if (this instanceof Array && this.length>0) {
        for (let i = 0; i < this.length; i++) {
            let itemres = itemfn(this[i], i)
            if(itemres) new_arr.push(itemres)
        }
    }
    return new_arr;
}
```
